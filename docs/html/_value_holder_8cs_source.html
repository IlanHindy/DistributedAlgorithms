677  public static bool BelongsToBaseAlgorithmEnum(this IValueHolder valueHolder, NetworkElement networkElement, NetworkElement.ElementDictionaries mainDictionary)
678  {
679  // If the value holder is not attribute get the attribute of the value holder
680  Attribute attribute;
681  if (!(valueHolder is Attribute))
682  {
683  attribute = (Attribute)valueHolder.Parent;
684  }
685  else
686  {
687  attribute = (Attribute)valueHolder;
688  }
689 
690  // The condition is that the attribute which is the ancestor of our attribute
691  // which is in the main dictionary belongs to the base class
692  // Find the ancestor.
693  // The parent sequence is ending in the following way
694  // null <- networkElement <- Attribute <- mainDictionary <- Attribute
695  // We have to find the last element in the chain described below
696  // We do it with a loop that advances 2 parameters: The attribute and the chain end
697  // The chain end is 4 times the parent of the attribute
698  // Get the first chain end (If we cannot end the chain that means that the valueHolder is
699  // one of the value holders at the middle of the chain hence they are not attributes in the
700  // base classes main dictionary
701  IValueHolder chainEnd = attribute;
702  for (int idx = 0; idx < 4; idx++)
703  {
704  chainEnd = chainEnd.Parent;
705  if (chainEnd == null && idx != 3)
706  {
707  return false;
708  }
709  }
710 
711  // Find the ancestor
712  while (chainEnd != null)
713  {
714  chainEnd = chainEnd.Parent.Parent;
715  attribute = (Attribute)attribute.Parent.Parent;
716  }

719  if (!((AttributeDictionary)networkElement[mainDictionary]).Values.Any(a => a == attribute))
720  {
