700  public static bool BelongsToBaseAlgorithmEnum(this IValueHolder valueHolder, NetworkElement networkElement, NetworkElement.ElementDictionaries mainDictionary)
701  {
702  // If the value holder is not attribute get the attribute of the value holder
703  Attribute attribute;
704  if (!(valueHolder is Attribute))
705  {
706  attribute = (Attribute)valueHolder.Parent;
707  }
708  else
709  {
710  attribute = (Attribute)valueHolder;
711  }
712 
713  // The condition is that the attribute which is the ancestor of our attribute
714  // which is in the main dictionary belongs to the base class
715  // Find the ancestor.
716  // The parent sequence is ending in the following way
717  // null <- networkElement <- Attribute <- mainDictionary <- Attribute
718  // We have to find the last element in the chain described below
719  // We do it with a loop that advances 2 parameters: The attribute and the chain end
720  // The chain end is 4 times the parent of the attribute
721  // Get the first chain end (If we cannot end the chain that means that the valueHolder is
722  // one of the value holders at the middle of the chain hence they are not attributes in the
723  // base classes main dictionary
724  IValueHolder chainEnd = attribute;
725  for (int idx = 0; idx < 4; idx++)
726  {
727  chainEnd = chainEnd.Parent;
728  if (chainEnd == null && idx != 3)
729  {
730  return false;
731  }
732  }
733 
734  // Find the ancestor
735  while (chainEnd != null)
736  {
737  chainEnd = chainEnd.Parent.Parent;
738  attribute = (Attribute)attribute.Parent.Parent;
739  }

743  {
