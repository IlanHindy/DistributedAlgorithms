567  private void HandlePresnpMessage(BaseMessage message)
568  {
569  int messageSource = message.GetHeaderField(BaseMessage.HeaderFieldKeys.SourceProcess).Value;
570  int messageRound = message.GetHeaderField(BaseMessage.HeaderFieldKeys.Round).Value;
571  int messageExpected = message.GetField(LayYoungMessage.FieldKeys.Expected).Value;
572  double messageWeight = message.GetField(LayYoungMessage.FieldKeys.Weight).Value;
573  int processRound = OperationResults.Value[OperationResultKeys.Round].Value;
574  int processParent = OperationResults.Value[OperationResultKeys.Parent].Value;
575  LayYoungChannel channel = (LayYoungChannel)InChannels().First(c => c[ElementDictionaries.ElementAttributes].Value[BaseChannel.ElementAttributeKeys.SourceProcess].Value == messageSource);
576  int channelCount = channel[ElementDictionaries.OperationResults].Value[LayYoungChannel.OperationResultKeys.Count].Value;
577  int channelPrevRoundCount = channel[ElementDictionaries.OperationResults].Value[LayYoungChannel.OperationResultKeys.PrevRoundCount].Value;
578 
579 
580  channel[ElementDictionaries.OperationResults].Value[LayYoungChannel.OperationResultKeys.Count].Value = 0;
581 
582  // Presnp that caused the snapshot
583  if (messageRound > processRound)
584  {
585  TakeSnapshot(messageWeight / 2, messageSource, channel);
586  channel[ElementDictionaries.OperationResults].Value[LayYoungChannel.OperationResultKeys.Weight].Value = messageWeight / 2;
587  channel[ElementDictionaries.OperationResults].Value[LayYoungChannel.OperationResultKeys.Expected].Value = messageExpected - channelPrevRoundCount;
588  }
589 
590  // Presnp that came after a snapshot was done (by base message)
591  else
592  {
593 
594  // If the message does not have weight - nothing has to be done - waiting for weight message
595  if (messageWeight > 0)
596  {
597  // If the presnap is from the parent - send the weight to all neighbours
598  if (processParent == messageSource)
599  {
600  SendWeight(messageWeight / 2);
601  channel[ElementDictionaries.OperationResults].Value[LayYoungChannel.OperationResultKeys.Weight].Value = messageWeight / 2;
602  channel[ElementDictionaries.OperationResults].Value[LayYoungChannel.OperationResultKeys.Expected].Value = messageExpected - channelPrevRoundCount;
603  }
604 
605  // If the message is from processor that did not couse the stapshot:
606  // update expected and weight of the channel
607  else
608  {
609  channel[ElementDictionaries.OperationResults].Value[LayYoungChannel.OperationResultKeys.Weight].Value = messageWeight;
610  channel[ElementDictionaries.OperationResults].Value[LayYoungChannel.OperationResultKeys.Expected].Value = messageExpected - channelCount;
611  }
612  channel[ElementDictionaries.OperationResults].Value[LayYoungChannel.OperationResultKeys.Count].Value = 0;
613  }
614  }
615  SendReportIfNeeded();

640  private void HandleReportMessage(BaseMessage message)
641  {
642  int messageRound = message.GetHeaderField(BaseMessage.HeaderFieldKeys.Round).Value;
643  int messageOriginator = message.GetField(LayYoungMessage.FieldKeys.Id).Value;
644  int messageSource = message.GetHeaderField(BaseMessage.HeaderFieldKeys.SourceProcess).Value;
645  double messageWeight = message.GetField(LayYoungMessage.FieldKeys.Weight).Value;
646  int processRound = OperationResults.Value[OperationResultKeys.Round].Value;
647  int maxRounds = PrivateAttributes.Value[PrivateAttributeKeys.MaxRounds].Value;
648  bool isOriginator = ElementAttributes.Value[ElementAttributeKeys.Initiator].Value;
649  Attribute originatorSnapshot = message.GetField(LayYoungMessage.FieldKeys.Snapshot);
650  Attribute originatorSnapshotMessages = message.GetField(LayYoungMessage.FieldKeys.SnapshotMessages);
651 
652  // If the message is from a finished round - exit
653  if (processRound > messageRound + 1)
654  {
655  return;
656  }
657 
658  // Se parameters according to whether the message is from this round or prev round
659  AttributeList reportSources;
660  Attribute weightAttribute;
661  bool messageFromPrevRound;
662  if (processRound == messageRound)
663  {
664  reportSources = OperationResults.Value[OperationResultKeys.CurrentRoundReportSources].Value;
665  weightAttribute = OperationResults.Value[OperationResultKeys.CurrentRoundWeight];
666  messageFromPrevRound = false;
667  }
668  else
669  {
670  reportSources = OperationResults.Value[OperationResultKeys.PrevRoundReportSources].Value;
671  weightAttribute = OperationResults.Value[OperationResultKeys.PrevRoundWeight];
672  messageFromPrevRound = true;
673  }
674 
675  // If this is not the first message from the origins process - exit
676  if (reportSources.Any(a => a.Value == messageOriginator))
677  {
678  return;
679  }
680 
681  // If this is the first message from the origin process - add the originator to the already arrived originators
682  else
683  {
684  reportSources.Add(new Attribute { Value = messageOriginator });
685  }
686 
687 
688  // If the process is an originator
689  if (isOriginator)
690  {
691  // Update the dictionaries that holds the result
692  AttributeDictionary networkSnapshots = OperationResults.Value[OperationResultKeys.NetworkSnapshots].Value;
693  networkSnapshots.Add(messageOriginator, new Attribute { Value = originatorSnapshot.Value });
694  AttributeDictionary networkSnapshotsMessages = OperationResults.Value[OperationResultKeys.NetworkSnapshotsMessages].Value;
695  networkSnapshotsMessages.Add(messageOriginator, new Attribute { Value = originatorSnapshotMessages.Value });
696  weightAttribute.Value += messageWeight;
697 
698  // If this is a message from the prev round
699  // Check if the round finished and decide whether to generate a message or
700  // start a termination process
701  if (messageFromPrevRound)
702  {
703  // Decide if this is the end of the round
704  // The end of the round is if the process weight + all network processes weights + channels wheights is 1
705  double totalWeight = weightAttribute.Value;
706  foreach (LayYoungChannel channel in IncommingChannels)
707  {
708  totalWeight += channel[ElementDictionaries.OperationResults].Value[LayYoungChannel.OperationResultKeys.Weight].Value;
709  }
710  if (totalWeight == 1)
711  {
712  ShowResults(messageRound);
713 
714  // If reached the last round - terminate
715  if (maxRounds == processRound)
716  {
717  Terminate();
718  }
719 
720  // If did not reach the last round - perform take snapshot - start a new round
721  else
722  {
723  OperationResults.Value[OperationResultKeys.NetworkSnapshots].Value = new AttributeDictionary();
724  OperationResults.Value[OperationResultKeys.NetworkSnapshotsMessages].Value = new AttributeDictionary();
725  TakeSnapshot(0.5, ElementAttributes.Value[NetworkElement.ElementAttributeKeys.Id].Value, null);
726  }
727  }
728  }
729  }
730 
731  // If the process is not an initiators - propagate the message to all neighbours except
732  // The process that this message was sent from and the originator of the message
733  else
734  {
735  SendToNeighbours(message, new List<int> { messageOriginator, messageSource });
736  }
737  }

785  private void TakeSnapshot(double weight, int messageSource, LayYoungChannel sourceChannel)
786  {
787 
788  // Keep the count (of the round that is about to finish) in PrevRoundCount
789  // for the case that the snapshot is done not by presnp
790  // This operation is not done in case the method is activated by the initiator
791  // (The sourceChannel == null)
792  if (sourceChannel != null)
793  {
794  sourceChannel[ElementDictionaries.OperationResults].Value[LayYoungChannel.OperationResultKeys.PrevRoundCount].Value =
795  sourceChannel[ElementDictionaries.OperationResults].Value[LayYoungChannel.OperationResultKeys.Count].Value;
796  sourceChannel[ElementDictionaries.OperationResults].Value[LayYoungChannel.OperationResultKeys.Count].Value = 0;
797  }
798 
799  // advance the round
800  Attribute roundAttribute = OperationResults.Value[OperationResultKeys.Round];
801  BaseAlgorithmUserEvent(roundAttribute.Value);
802  roundAttribute.Value++;
803 
804  // Take the snapshot
805  OperationResults.Value[OperationResultKeys.Snapshot].Value = roundAttribute.Value;
806 
807  // set the parent
808  OperationResults.Value[OperationResultKeys.Parent].Value = messageSource;
809 
810  // Send presnp to all the neighbours
811  SendPresnp(weight, roundAttribute.Value);
812 
813 
814  // Clear the weight of the prev round
815  foreach (LayYoungChannel channel in InChannels())
816  {
817  channel[ElementDictionaries.OperationResults].Value[LayYoungChannel.OperationResultKeys.Weight].Value = 0;
818  channel[ElementDictionaries.OperationResults].Value[LayYoungChannel.OperationResultKeys.State].Value = new AttributeList();
819  }
820 
821  // Handle Report Data
822  ReportHandelingDataStartRound();
823  }
