366  public override void ReceiveHandling(BaseMessage message)
367  {
368  ChandyLamportChannel channel = (ChandyLamportChannel)ChannelFrom(message);
369 
370  switch (message[bm.pak.MessageType])
371  {
372  case m.MessageTypes.BaseMessage:
373 
374  // If the process performed snapshot but still did not get marker from all
375  // it's neighbors :
376  // Save the message - It will be sent as message in the channels
377  if (or[p.ork.Recordered] && !channel.or[c.ork.Marker])
378  {
379  channel.or[c.ork.State].Add(message[bm.ork.MessageName]);
380  }
381  break;
382  case m.MessageTypes.Marker:
383 
384  // If the process received a marker:
385  // Add the weight in the marker
386  // Perform TakeSnapshot (If first Marker in round - Send Marker to all the neighbors)
387  or[p.ork.Weight] += message[m.marker.Weight];
388  TakeSnapshot();
389 
390  // Check if the round ended (Received Marker from all it's neighbors)
391  // Perform EndSnapshot (Send Report, reset variables)
392  channel.or[c.ork.Marker] = true;
393  if (InChannels.All(cnl => cnl.or[c.ork.Marker]))
394  {
395  EndSnapshot();
396  }
397  break;
398  case m.MessageTypes.Report:
399 
400  // If received a Report Message
401  // If this is not the first report from the source processor in this round
402  // throw the message
403  if (message[bm.pak.Round] < or[bp.ork.Round] ||
404  ((AttributeList)or[p.ork.ReceivedMessageFrom]).Any((a => a.Value == (int)message[m.report.Id])))
405  {
406  break;
407  }
408  else
409  {
410  or[p.ork.ReceivedMessageFrom].Add(message[m.report.Id]);
411  }
412 
413  // If the process is the initiator
414  // Add the message to the results
415  // Add the weight to the process weight
416  // Check condition for end round (weight == 1)
417  // Check condition for end running (round = max rounds)
418  if (ea[bp.eak.Initiator])
419  {
420  or[p.ork.Results].Add(message[m.report.Snapshot]);
421  or[p.ork.Weight] += message[m.report.Weight];
422  pp[bp.ppk.Text] = GetProcessDefaultName() + "\n" + or[p.ork.Weight];
423  if (or[p.ork.Weight] == 1)
424  {
425  PrintResults();
426 
427  if (or[bp.ork.Round] < pa[p.pak.MaxRounds])
428  {
429  TakeSnapshot();
430  }
431  else
432  {
433  Terminate();
434  }
435  }
436  }
437 
438  // If the process is not the initiator
439  // Propagate the message to all the neighbors
440  else
441  {
442  SendToNeighbours(message, SelectingMethod.Exclude, new List<int> { (int)message[m.report.Id] });
443  }
444  break;
445  }
446  }

733  private void TakeSnapshot()
734  {
735  if (!or[p.ork.Recordered])
736  {
737  or[bp.ork.Round]++;
738  or[p.ork.Snapshot] = or[bp.ork.Round];
739  or[p.ork.Recordered] = true;
740  or[p.ork.ReceivedMessageFrom].Clear();
741  double weight = or[p.ork.Weight];
742  int numOfNeighbours = OutChannels.Count;
743  or[p.ork.Weight] = weight / 2;
744  SendMarker(MessageDataFor_Marker(bm.PrmSource.Prms, null,
745  weight / (2 * numOfNeighbours)));
746  pp[bp.ppk.Text] = GetProcessDefaultName() + "\n" + or[p.ork.Weight];
747 
748  }
749  }

768  private void EndSnapshot()
769  {
770 
771  // If the process is the initiator - save the snapshot to the results
772  if (ea[bp.eak.Initiator])
773  {
774  or[p.ork.Results].Add(RecordSnapshot());
775  }
776 
777  // If the process is not the initiator - Send a report and clear the weight
778  else
779  {
780  SendReport(MessageDataFor_Report(bm.PrmSource.Prms, null, ea[ne.eak.Id], RecordSnapshot(), or[p.ork.Weight]));
781  or[p.ork.Weight] = 0;
782  }
783 
784  // Set the self Id to the list of source processes (in order to block report message coming
785  // from this processor (If it is in a circle)
786  or[p.ork.ReceivedMessageFrom].Add(ea[ne.eak.Id]);
787 
788  // Init the algorithm flags and variables for the next round snapshot
789  or[p.ork.Recordered] = false;
790  for (int idx = 0; idx < InChannels.Count; idx++)
791  {
792  InChannels[idx].or[c.ork.Marker] = false;
793  InChannels[idx].or[c.ork.State].Clear();
794  }
795  }
