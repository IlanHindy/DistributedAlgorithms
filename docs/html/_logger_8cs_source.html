24  * Class Logger
25  * This class handles log files and error messages
26  * The following is the features of this class
27  * 1. This class handle a main log file that receives logs from all the objects.
28  * 2. This class handles a separate log file receives logs according to the fields
29  * 3. Multipile threads can send logs to one file
30  * 4. The log methods have a filter parameter according which the class can decide if to write
31  * or not to write the log
32  * 5. The log methods have a LogMode parameter that handle the targets of the logs
33  * 5.1 Whether to write the log to the main log file
34  * 5.2 Whether to write the log to the process log file
35  * 5.3 Whether to generate a CustomizedMessageBox with the log
36  * 6. The following are the log file name conventions
37  * 6.1 The main log file name is composed from 3 parts : path + name + "log"
38  * 6.2 The process log file is composed from 4 parts : path + main log name + process name + "log
39  * 7. There are 2 main log functions
40  * 7.1 log a string
41  * 7.2 log an XmlDocument object
42  */

482  private static void WriteLog(LogMode logMode, string sourceObject, string log, string messageTraceLog, string logFilter)
483  {
484  lock (WriteLogLockObject)
485  {
486  //Find Whether to log according to the filter
487  //If the Filters list is empty write all the logs othere wise the logFilter parameter has
488  //to be found in the Filters list in order to write the log
489  if (Filters.Count > 0)
490  {
491  if (Filters.FirstOrDefault(s => s == logFilter) is null)
492  {
493  return;
494  }
495  }
496 
497  //Write to the main log file
498  if (traceLogFile == null)
499  {
500  OpenMainLogFile();
501  }
502  traceListener.WriteLine(log);
503  traceListener.Flush();
504 
505  //Write to the process private log file
506  if (logMode == LogMode.MainLogAndProcessLog ||
507  logMode == LogMode.MainLogProcessLogAndError ||
508  logMode == LogMode.MainLogAndProcessLogAndMessageTrace)
509  {
510  TextWriterTraceListener processTraceListener = GetProcessTraceListener(sourceObject);
511  processTraceListener.WriteLine(log);
512  processTraceListener.Flush();
513  }
514 
515  //Create error message if needed
516  if (logMode == LogMode.MainLogAndError ||
517  logMode == LogMode.MainLogProcessLogAndError)
518  {
519  MessageRouter.MessageBox(new List<String> { log },"Log Report", null, Icons.Error);
520  }
521  }
522 
523  //Write to message trace window
524  if (logMode == LogMode.MainLogAndProcessLogAndMessageTrace || logMode == LogMode.MainLogAndMessageTrace)
525  {
526  MessageRouter.ReportMessage(sourceObject, VectorClockString(), messageTraceLog);
527  }
528  }
