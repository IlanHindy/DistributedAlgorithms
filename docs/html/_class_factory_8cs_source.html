907  public static AttributeDictionary MessageDictionary(dynamic messageKey, NetworkElement messageDemo, BaseProcess process)
908  {
909  // Add an empty dictionary to the OperationResults
910  messageDemo.or.Add(messageKey, new Attribute { Value = new AttributeDictionary() });
911 
912  // Generate the method name
913  string methodName = ((AttributeDictionary)messageDemo.or[messageKey]).MessageMethodName();
914 
915  // Generate the enum name of the message. Note that Enum names begin with small letter and
916  // Enum values begin with big letter so there is a need to set the first letter to lower
917  // In order to get the enum name
918  string keyString = TypesUtility.GetKeyToString(messageKey);
919  keyString = char.ToLower(keyString[0]) + keyString.Substring(1);
920 
921  // The method for getting the data is invoked by it's name. When invoking a method in this
922  // way the exact number of parameters has to be given. Each method has a different number of parameters
923  // (according to the number of entries in the message enum so this number has to be retrieved
924  // from the enum
925  int optionalPrmsCount = Enum.GetValues(GenerateMessageEnum(keyString)).Length;
926 
927  // Activating the method
928  return (AttributeDictionary)TypesUtility.InvokeMethod(process, methodName, new List<object> { bm.PrmSource.Default, null }, optionalPrmsCount, true);
929  }
