931  public static AttributeDictionary MessageDictionary(dynamic messageKey, NetworkElement messageDemo, BaseProcess process)
932  {
933  // Add an empty dictionary to the OperationResults
934  messageDemo.or.Add(messageKey, new Attribute { Value = new AttributeDictionary() });
935 
936  // Generate the method name
937  string methodName = ((AttributeDictionary)messageDemo.or[messageKey]).MessageMethodName();
938 
939  // Generate the enum name of the message. Note that Enum names begin with small letter and
940  // Enum values begin with big letter so there is a need to set the first letter to lower
941  // In order to get the enum name
942  string keyString = TypesUtility.GetKeyToString(messageKey);
943  keyString = char.ToLower(keyString[0]) + keyString.Substring(1);
944 
945  // The method for getting the data is invoked by it's name. When invoking a method in this
946  // way the exact number of parameters has to be given. Each method has a different number of parameters
947  // (according to the number of entries in the message enum so this number has to be retrieved
948  // from the enum
949  int optionalPrmsCount = Enum.GetValues(GenerateMessageEnum(keyString)).Length;
950 
951  // Activating the method
952  return (AttributeDictionary)TypesUtility.InvokeMethod(process, methodName, new List<object> { bm.PrmSource.Default, null }, optionalPrmsCount, true);
