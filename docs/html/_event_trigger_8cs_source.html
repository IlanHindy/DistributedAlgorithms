219  private bool IsTrue(EventTriggerType trigger, BaseMessage message)
220  {
221  // Get the data of the message that caused the event
222  int messageRound = message.GetHeaderField(bm.pak.Round);
223  dynamic messageType = message.GetHeaderField(bm.pak.MessageType);
224 
225  // Get the other end of the message that caused the event:
226  // For a sending event this is the destination of the message
227  // For a receive event this is the source of the message
228  int messageOtherEnd;
229  if (trigger == EventTriggerType.AfterSendMessage || trigger == EventTriggerType.BeforeSendMessage)
230  {
231  messageOtherEnd = message.GetHeaderField(bm.pak.DestProcess);
232  }
233  else
234  {
235  messageOtherEnd = message.GetHeaderField(bm.pak.SourceProcess);
236  }
237 
238  if (this[Comps.Trigger] == trigger &&
239  this[Comps.Round] == messageRound &&
240  TypesUtility.CompareDynamics(this[Comps.MessageType], messageType) &&
241  this[Comps.OtherEnd] == messageOtherEnd)
242  {
243  return true;
244  }
245  return false;
246  }
