1007  protected void ArrangeItemsList(ItemsControl item)
1008  {
1009  /*
1010  * Initialize status:
1011  * The items list of the item is not updated
1012  * The controlsAttributeLinks is updated with the action that has to be done:
1013  * - Controled that where scanned (That meens that they are in the NetworkElement) will be inserted in the init list
1014  * - Controls that where not scanned but has update status != NotUpdated will be inserted according to theire location
1015  * - Controls that where not scanned and not updated that will be removed
1016  */
1017 
1018 
1019  /*
1020  * The Creation of the new items list is done in the following way
1021  * Create a list of all the items that where scanned
1022  * Create a list of all the items that where not scanned but cannot be removed because of update status
1023  * Order the list of items that where scanned or added.
1024  * insert the items of this list as chiled items of the main item
1025  * insert the updated items in the location that they have in the link
1026  */
1027 
1028  /*
1029  * Create a list of links that represent scanned items to insert and order it
1030  */
1031  List<ControlsAttributeLink> itemsToInsert = controlsAttributeLinks.Where(
1032  entry => entry.Value.wasScannedInTheLastScan == true && item.Items.IndexOf(entry.Key) != -1)
1033  .Select(entry => entry.Value).ToList();
1034  itemsToInsert = itemsToInsert.OrderBy(entry => entry.indexInPerant).ToList();
1035 
1036  /*
1037  * Create a list of all the updated but not scanned items
1038  */
1039  List<ControlsAttributeLink> updatedButNotScannedItems = (List<ControlsAttributeLink>)(controlsAttributeLinks.Where(
1040  entry => entry.Value.wasScannedInTheLastScan == false &&
1041  entry.Value.updateStatus != UpdateStatuses.NotUpdated &&
1042  item.Items.IndexOf(entry.Value.item) != -1)
1043  .Select(entry => entry.Value)
1044  .OrderBy(link => link.indexInPerant)
1045  .ToList());
1046 
1047  /*
1048  * Remove from the link dictionary the items that where not updated and not scanned
1049  */
1050  controlsAttributeLinks = controlsAttributeLinks.Where(
1051  entry => ConditionToStayInTheLinks(entry.Value, item))
1052  .ToDictionary(entry => entry.Key, entry => entry.Value);
1053 
1054  /*
1055  * Insert the scanned or added items as chiled items to the item
1056  */
1057  item.Items.Clear();
1058  itemsToInsert.ForEach(link => item.Items.Add(link.item));
1059 
1060  /*
1061  * insert the updated but not scaned items in the place that they have in the record
1062  * (before the last scan)
1063  */
1064  updatedButNotScannedItems.ForEach(link => item.Items.Insert(link.indexInPerant, link.item));
1065 
1066  /*
1067  * Update the links with the index from the item
1068  */
1069  UpdateIndexInParent(item);
1070 
1071  }
