348  public void PositionChannel()
349  {
350  // Get the processes circles controls
351  Ellipse sourceProcessPresentationCircle = GetProcessPresentationCircle(bc.eak.SourceProcess);
352  Ellipse destProcessPresentationCircle = GetProcessPresentationCircle(bc.eak.DestProcess);
353 
354  //Calculate the center of the circles
355  double sourceX0 = sourceProcessPresentationCircle.Margin.Left + sourceProcessPresentationCircle.Width / 2;
356  double sourceY0 = sourceProcessPresentationCircle.Margin.Top + sourceProcessPresentationCircle.Height / 2;
357  double destX0 = destProcessPresentationCircle.Margin.Left + destProcessPresentationCircle.Width / 2;
358  double destY0 = destProcessPresentationCircle.Margin.Top + destProcessPresentationCircle.Height / 2;
359 
360  //Calculate the direction of the line from the center this is used in order to select from
361  //the 2 possible pointes on the circle border that are connected to the line
362  double sourceXDirection = (sourceX0 < destX0) ? 1 : -1;
363  double destXDirection = (sourceX0 > destX0) ? 1 : -1;
364  double sourceYDirection = (sourceY0 < destY0) ? 1 : -1;
365  double destYDirection = (sourceY0 > destY0) ? 1 : -1;
366  double lineSourceX;
367  double lineSourceY;
368  double lineDestX;
369  double lineDestY;
370 
371  //Calculate the circle radioses
372  double sourceRadiouse = sourceProcessPresentationCircle.Width / 2;
373  double destRadiouse = destProcessPresentationCircle.Width / 2;
374 
375 
376  //If the source and destination X coordinate are the same
377  //(The source and the destination circles are one above the other and
378  // the line incline will be infinit
379  if (sourceX0 == destX0)
380  {
381  lineSourceX = sourceX0;
382  lineDestX = destX0;
383  lineSourceY = sourceY0 + sourceYDirection * sourceRadiouse;
384  lineDestY = destY0 + destYDirection * destRadiouse;
385  }
386  else
387  {
388  double lineIncline = (destY0 - sourceY0) / (destX0 - sourceX0);
389  lineSourceX = sourceXDirection * (Math.Sqrt(Math.Pow(sourceRadiouse, 2) / (Math.Pow(lineIncline, 2) + 1))) + sourceX0;
390  lineDestX = destXDirection * (Math.Sqrt(Math.Pow(destRadiouse, 2) / (Math.Pow(lineIncline, 2) + 1))) + destX0;
391  lineSourceY = lineIncline * (lineSourceX - sourceX0) + sourceY0;
392  lineDestY = lineIncline * (lineDestX - destX0) + destY0;
393  }
394 
395  //Position the poligons (The arrow head)
396  ((Line)additionalControls[AdditionalControlKeys.Line]).X1 = lineSourceX;
397  ((Line)additionalControls[AdditionalControlKeys.Line]).Y1 = lineSourceY;
398  ((Line)additionalControls[AdditionalControlKeys.Line]).X2 = lineDestX;
399  ((Line)additionalControls[AdditionalControlKeys.Line]).Y2 = lineDestY;
400 
401  //If the line start or end are inside the circles - set the status of the presentation
402  //To do not show else the status is NotSelected - call to a function that will select
403  //And set the colors of the controls
404  if (InOtherCircle(lineSourceX, lineSourceY, destX0, destY0, destRadiouse) ||
405  InOtherCircle(lineDestX, lineDestY, sourceX0, sourceY0, sourceRadiouse))
406  {
407  SetChannelsStatuses(MainWindow.SelectedStatus.DoNotShow);
408  }
409  else
410  {
411  SetChannelsStatuses(MainWindow.SelectedStatus.NotSelected);
412  }
413 
414  //Position the poligons (The arrow head)
415  double[] lineX = { lineDestX, lineSourceX };
416  double[] lineY = { lineDestY, lineSourceY };
417  double angle = GetAngle(lineSourceX, lineSourceY, lineDestX, lineDestY);
418  double[] angels = { 180 + angle, angle };
419  for (int idx = 0; idx < presentationElements.Count; idx++)
420  {
421  PositionPolygon((Polygon)presentationElements.Values.ElementAt(idx).controls[PresentationElement.ControlKeys.ArrowHead], lineX[idx], lineY[idx], angels[idx]);
422  PositionMessagesGrid((GridView)presentationElements.Values.ElementAt(idx).controls[PresentationElement.ControlKeys.MessagesGridView], labelRelativePosition[idx]);
423  //PositionLabel((Label)presentationElements.Values.ElementAt(idx).controls[PresentationElement.ControlKeys.MessageQ], labelRelativePosition[idx]);
424  }

427  if (additionalControls.ContainsKey(AdditionalControlKeys.LabelsPanel))
428  {
429  PositionBorder((Border)additionalControls[AdditionalControlKeys.LabelsBorder], 0.5);
430  }
431  }

528  /*
529  * Get the angle of a line
530  */
531 
532  /**********************************************************************************************/
547  private double GetAngle(double x1, double y1, double x2, double y2)
548  {
549  double degrees;
550 
551  // Avoid divide by zero run values.
552  if (x2 - x1 == 0)
553  {
554  if (y2 > y1)
555  degrees = 90;
556  else
557  degrees = 270;
558  }
559  else
560  {
561  // Calculate angle from offset.
562  double riseoverrun = (double)(y2 - y1) / (double)(x2 - x1);
563  double radians = Math.Atan(riseoverrun);
564  degrees = radians * ((double)180 / Math.PI);
565 
566  // Handle quadrant specific transformations.
567  if ((x2 - x1) < 0 || (y2 - y1) < 0)
568  degrees += 180;
569  if ((x2 - x1) > 0 && (y2 - y1) < 0)
570  degrees -= 180;
571  if (degrees < 0)
572  degrees += 360;
573  }
574  return degrees;
575  }
