508  private void ReportArrived(BaseMessage report)
509  {
510  // Get attributes of the process
511  AttributeDictionary receivedReportFrom = OperationResults.Value[OperationResultKeys.ReceivedReportFrom].Value;
512  bool initiator = ElementAttributes.Value[BaseProcess.ElementAttributeKeys.Initiator].Value;
513  int maxRound = PrivateAttributes.Value[PrivateAttributeKeys.NumberOfRounds].Value;
514  int processId = ElementAttributes.Value[NetworkElement.ElementAttributeKeys.Id].Value;
515  int processSnapshot = OperationResults.Value[OperationResultKeys.Snapshot].Value;
516  int processRound = OperationResults.Value[OperationResultKeys.RoundNum].Value;
517  double processWeight = OperationResults.Value[OperationResultKeys.Weight].Value;
518  AttributeDictionary processMessagesArived = OperationResults.Value[OperationResultKeys.MessagesReceived].Value;
519 
520  // Get attributes of the message
521  double messageWeight = report.GetField(ChandyLamportMessage.FieldKeys.Weight).Value;
522  int messageId = report.GetField(ChandyLamportMessage.FieldKeys.Id).Value;
523  int messageSnapshot = report.GetField(ChandyLamportMessage.FieldKeys.Snapshots).Value;
524  int messageRound = report.GetHeaderField(BaseMessage.HeaderFieldKeys.Round).Value;
525  AttributeList listOfMessagesArrived = report.GetField(ChandyLamportMessage.FieldKeys.MessageSnapshot).Value;
526 
527  // The algorithm
528  // If a message from the process arrived (in this round)
529  // or the round advanced (which mees that the initiator got report from al the processor
530  // throw the message
531  if (!receivedReportFrom.Keys.Any(key => key == messageId) && processRound == messageRound)
532  {
533  receivedReportFrom.Add(messageId, new Attribute { Value = messageSnapshot });
534  processMessagesArived.Add(messageId, new Attribute { Value = listOfMessagesArrived });
535  if (initiator)
536  {
537  processWeight += messageWeight;
538  if (processWeight == 1)
539  {
540  // Add the initiator's snapshot to the snapshot list
541  receivedReportFrom.Add(processId, new Attribute { Value = processSnapshot });
542  // Collect the messages from the channels
543  AttributeList messages = new AttributeList();
544  foreach (ChandyLamportChannel channel in InChannels())
545  {
546  messages.Concat(channel.ReportChannelMessages());
547  }
548  processMessagesArived.Add(processId, new Attribute { Value = messages });
549 
550  // Report the snapshots
551  PrintSnapshot(processRound, receivedReportFrom, processMessagesArived);
552 
553  // Start a new round
554  processRound++;
555  if (processRound < maxRound)
556  {
557  InitNewRound(processRound);
558  }
559  else
560  {
561  Terminate();
562  }
563  }
564  else
565  {
566  // Set the attributes for the initiator
567  OperationResults.Value[OperationResultKeys.Weight].Value = processWeight;
568  OperationResults.Value[OperationResultKeys.ReceivedReportFrom].Value = receivedReportFrom;
569  }
570  }
571  else
572  {
573  SendToNeighbours(report, null);
574  OperationResults.Value[OperationResultKeys.ReceivedReportFrom].Value = receivedReportFrom;
575  }
576  }

605  private void TakeSnapshot(BaseMessage message)

608  double weight = message.GetField(ChandyLamportMessage.FieldKeys.Weight).Value;
609  int round = message.GetHeaderField(BaseMessage.HeaderFieldKeys.Round).Value;
