419  private void CheckBackwordChannel(List<BuildCorrectionParameters> buildCorrectionsParameters)
420  {
421  if (network.ea[bn.eak.DirectedNetwork] == true)
422  {
423  return;
424  }
425  else
426  {
427  //Check if the channel has a backwored channel
428  //Exclude the self connecting channels
429  if (ea[bc.eak.SourceProcess] == ea[bc.eak.DestProcess])
430  {
431  return;
432  }
433  if (CheckIfBackwordChannelNeeded(network))
434  {
435  BuildCorrectionParameters newCorrection = new BuildCorrectionParameters();
436  newCorrection.ErrorMessage = "There is no backwored channel for channel : " + ToString();
437  newCorrection.CorrectionMethod = CreateBackworedChannel;
438  newCorrection.CorrectionParameters = network;
439  buildCorrectionsParameters.Add(newCorrection);
