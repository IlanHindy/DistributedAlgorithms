46  /*
47  * Description of the algorithm:
48  * Data for each processor:
49  * - Parent process id
50  * - Number of childrens
51  * The algorithm creates a dynamic tree which increases and decreases.
52  * The conditions to enter the tree are:
53  * - If the process is initiator (It did not receive a message before it's first send)
54  * The process is added as a root for the tree When it sends it's first message
55  * (There can be many trees simultanissaly)
56  * - If the process is not an initiator
57  * The process is added to the tree if it is not in the tree already when it
58  * received it's first message
59  * Handling the number of childrens
60  * - When a process sends message it increases the number of the childrens
61  * - When a process receives a message
62  * if it is in the tree it decreases the number of childres in the sending process
63  * if it is not in the tree it enters to the tree and the number of childrens in the sending
64  * process stays the same
65  * The conditions to leave the tree are
66  * - If the process does not have childrens
67  * - If the process stopped
68  * If these conditions are fullfiled the process leaves the tree
69  * When a process leaves the tree it decreases the number of childrens in the parent process
70  * and the parent process trys to leave the tree according to the same conditions
71  */

165  * CheckTermination
166  * There are 2 conditions for removing from the tree:
167  * 1. There are no messages or the breakpoint evaluation returned true
168  * 2. The number of childes is 0
169  * The operation of this method is done:
170  * 1. Every time that the process enters the receive loop iteration (because it's status might be stopped)

172  * 3. Evry time the number of childrens reached 0
173  * If these conditions fullfiled the following has to be done
174  * 1. If the process is initiator
175  * 1.1 remove the record from the dictionary

197  public bool CheckTermination(BaseProcess process, BaseNetwork network)
198  {
199 
200  AttributeList messageQueue = process.or[bp.ork.MessageQ];
201  int processId = process.ea[ne.eak.Id];
202 
203  //If the list is empty return success
204  if (processesData.Count == 0) return true;
205 
206  //If the process has no record - this is the beginning of a first iteration - return false
207  if (!processesData.ContainsKey(processId)) return false;
208 
209  int parentId = processesData[processId].parentId;
210 
211  //Checking the conditions
212  //1. If the message queue is empty the process stopped
213  //2. If there is a true breakpoint the process stopped
214  //3. If the process did not stop return false
215  if (!(messageQueue.Count == 0 || process.WaitingForBreakpointEvent == true)) return false;
216 
217  //Check if the process has no children
218  if (processesData[processId].numberOfChilds > 0) return false;
219 
220  //If the process finished remove it from the dictionary
221  processesData.Remove(processId);
222 
223  //Check if all the processes terminated
224  if (processesData.Count == 0) return true;
225 
226  //If this is not a root (and the process finished) check if the parent should terminate
227  if (parentId != -1)
228  {
229  processesData[parentId].numberOfChilds --;
230  BaseProcess parentProcess = network.Processes.First(p => p.ea[ne.eak.Id] == parentId);
231  return CheckTermination(parentProcess, network);
232  }
233  return false;
