642  private void RunAlgorithmEnvelope()
643  {
644  // Init the message queue (the message parameter is for comaptibility and not realy inserted
645  // to the queue)
646  BaseMessage message = new BaseMessage(network);
647  MessageQHandling(ref message, MessageQOperation.Init);
648 
649  // Create the listen thread to listen for new connections
650  CreateListenThread();
651 
652  if (InDebugMode)
653  {
654  // Sends the messages that where sent from this processor in the previouse
655  // activation and where waiting in the dest's Q when the running was stopped
656  SendInitDebugMessages();
657  }
658  else
659  {
660  // If the process is initiator activate the RunAlgorithm method
661  // to initiate the algorithm
662  if (ea[bp.eak.Initiator] == true)
663  {
664  // There are 3 ways to start the running:
665  // RunToEnd - Run the algorithm untill it ends or reached a breakpoint
666  // SingleStepAll - Activate the initiators
667  // InitializeRunning - Wait for the mouse press to activate the initiators
668  // In order to pause the network a breakpoint is used
669  // The value of the breakpoint is true for the first 2 initiation methods
670  // The value of the breakpoint is false for the thired initiation method
671  // So only in the thired initiation method the running will halt before activating the initiator
672  WaitingForBreakpointEvent = true;

674  WaitingForBreakpointEvent = false;

676  RunAlgorithm();

737  private void ReceiveHandlingEnvelope()
738  {
739  while (true)
740  {
741  // Retrieve the first message from the Queue
742  // This method will return only when there is a message in the queue
743  MessageQHandling(ref MessageInProcess, MessageQOperation.RetrieveFirstMessage);
744 
745  // Breakpoint handeling
746  // The following parameter is set when the breakpoints are addedqupdated from the gui
747  BreakpointsWhereUpdated = true;
748  while (BreakpointsWhereUpdated)
749  {
750  BreakpointsWhereUpdated = false;
751 
752  // Evaluate the breakpoints assigned to the :
753  // Network
754  // This process
755  // The process channels
756  // The message
757  trueBreakpoints = EvaluateBreakpoints(MessageInProcess);
758  WaitingForBreakpointEvent = trueBreakpoints.Count > 0;
759 
760  // The following method will cause the RunningHandler to check if a processing step
761  // was finished. The method is sent from here because this is the place that
762  // the process thread finishes a single step
763  MessageRouter.CheckFinishProcessingStep(this, new object[] { });
764 
765  if (WaitingForBreakpointEvent)
766  {
767  // hold processing untill the event will be set from the RunningHandler
768  BreakpointEvent.Reset();
769  }
770  else
771  {
772  // continue processing
773  BreakpointEvent.Set();
774  }
775  BreakpointEvent.WaitOne();
776  WaitingForBreakpointEvent = false;
777  }
778 
779  // Reset the list of breakpoints that returned true
780  trueBreakpoints = new List<Breakpoint>();
781 
782  // An empty message is used when forcing a termination
783  // and the message in processes needs to be thrown
784  // An empty message is used also when the message processing condition is false
785  if (MessageInProcess.IsEmpty())
786  {
787  continue;
788  }
789  else
790  {
791  MessageQHandling(ref MessageInProcess, MessageQOperation.RemoveFirstMessage);
792  }
793 
794  //A "Terminate" message is used to close the thared and the scket
795  if (TypesUtility.CompareDynamics(MessageInProcess.GetHeaderField(bm.pak.MessageType), bm.MessageTypes.Terminate))
796  {
797 
798  // The method Terminate returns true if the process finished the termination algorithm and should stop
799  if (Terminate(MessageInProcess))
800  {
801  MessageRouter.CheckFinishProcessingStep(this, new object[] { });
802  return;
803  }
804  }
805  else
806  {
807  // Activate the ReceiveHandling method of the process (Activate the process algorithm)
808  // See if there is a need to send Base algorithm message
809  ProcessEvents(EventTriggerType.BeforeReceiveMessage, MessageInProcess);
810 
811  // This is the method that the user's algorithm implement
812  ReportText("Processing : " + MessageInProcess.ShortDescription());
813  ReceiveHandling(MessageInProcess);
814 
815  // See if there is a need to send Base Algorithm message
816  ProcessEvents(EventTriggerType.AfterReceiveMessage, MessageInProcess);
817 
818  // Check if a processing step finished
819  MessageRouter.CheckFinishProcessingStep(this, new object[] { });
820  }

985  virtual public void ReceiveHandling(BaseMessage message)
986  {
987  int minProcessId = network.Processes.Min(process => process.ea[ne.eak.Id]);
988  int maxProcessId = network.Processes.Max(process => process.ea[ne.eak.Id]);
989 
990  // If the message is not Base message throw it
991  dynamic messageType = message.GetHeaderField(bm.pak.MessageType);
992  if (!(TypesUtility.CompareDynamics(messageType, bm.MessageTypes.Forewared) ||
993  TypesUtility.CompareDynamics(messageType, bm.MessageTypes.LastMessage)))
994  {
995  return;
996  }
997 
998  // If the process is the process with the maxProcessId (used to indicate the non-initiator)
999  if (ea[ne.eak.Id] == maxProcessId)
1000  {
1001  // Get the round of the message
1002  int messageCounter = message.GetHeaderField(bm.pak.Round);
1003 
1004  // Get the outgoing channel toword the process with of the minProcessId
1005  BaseChannel channel = OutGoingChannels.First(c => c.ea[bc.eak.DestProcess] == minProcessId);
1006 
1007  // If this is round 2 - Terminate
1008  if (messageCounter == 5)
1009  {
1010  BaseMessage newMessage = new BaseMessage(network,
1011  bm.MessageTypes.LastMessage,
1012  channel, "",
1013  messageCounter,
1014  minProcessId);
1015  Send(0, newMessage);
1016  Terminate();
1017  }
1018 
1019  // If this is not the end of round 2 - start a new round
1020  else
1021  {
1022  messageCounter++;
1023  BaseMessage newMessage = new BaseMessage(network, bm.MessageTypes.Forewared, channel, "", messageCounter, minProcessId);
1024  Send(minProcessId, newMessage);
1025  }
1026  }
1027 
1028  // Process with minProcessId
1029  else
1030  {
1031 
1032  // If a Terminate message arrived - Terminate()
1033  if (TypesUtility.CompareDynamics(message.GetHeaderField(bm.pak.MessageType),bm.MessageTypes.LastMessage))
1034  {
1035  // The termination will be recieved by all the processes in the network
1036  Terminate();
1037  }
1038 
1039  // If an algorithm message arrived - send it back
1040  else
1041  {
1042  BaseChannel channel = OutGoingChannels.First(c => c.ea[bc.eak.DestProcess] == maxProcessId);
1043  BaseMessage newMessage = new BaseMessage(network,
1044  bm.MessageTypes.Forewared,
1045  channel,
1046  message.GetField(bm.ork.Name),
1047  message.GetHeaderField(bm.pak.Round), maxProcessId);
