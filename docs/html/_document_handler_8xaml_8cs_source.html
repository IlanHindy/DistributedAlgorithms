1101 
1102  // The algorithm for filtering
1103  // For each paragraph in the destination:
1104  // Search (starting from the previous paragraph found) for the paragraph
1105  // if found : set all the paragraphs from the last found to the new found position collapsed
1106  int firstSeachIdx = 0; // This variable holds the place of the end of the previous search
1107  int paragraphSearchIdx; // This variable holds the index of the current search
1108  foreach (OpenXmlElement paragraph in destFileParagraphs)
1109  {
1110  string style = GetParagraphStyleId(paragraph);
1111  string text = paragraph.InnerText;
1112  if (!(style == "" && text == ""))
1113  {
1114 
1115  // Search for the destination paragraph in the source paragraphs starting from the last paragraph that was found
1116  for (paragraphSearchIdx = firstSeachIdx; paragraphSearchIdx < DockPanel_Paragraphs.Children.Count; paragraphSearchIdx++)
1117  {
1118  Grid grid = (Grid)DockPanel_Paragraphs.Children[paragraphSearchIdx];
1119  string presentationParagraphText = ((TextBlock)grid.Children[0]).Text;
1120  string presentationParagraphStyle = ((TextBlock)grid.Children[2]).Text;
1121 
1122  // If the paragraph was found set all the paragraphs from the last found to the current found - collapsed
1123  if (presentationParagraphStyle == style && presentationParagraphText == text)
1124  {
1125  for (; firstSeachIdx < paragraphSearchIdx; firstSeachIdx++)
1126  {
1127  ((Grid)DockPanel_Paragraphs.Children[firstSeachIdx]).Visibility = Visibility.Collapsed;
1128  }
1129  paragraphSearchIdx++;
1130  firstSeachIdx++;
1131  break;
1132  }
1133  }
1134  }
1135  }
1136 
1137  // Delete all the paragraphs from the last found to the end
1138  for (; firstSeachIdx < DockPanel_Paragraphs.Children.Count; firstSeachIdx++)
1139  {
1140  ((Grid)DockPanel_Paragraphs.Children[firstSeachIdx]).Visibility = Visibility.Collapsed;
1141  }

1638  break;
