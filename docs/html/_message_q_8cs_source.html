498  public void ChangeOrder(List<int> permutations, int sourceProcessId)
499  {
500  // Create a list of indexes of the messages coming from the source process
501  // This list is used to convert indexes from the permutation list to the
502  // MessageQues :
503  // If we have a value in the permutations the message that corresponds to this value
504  // Is channelMessagesIdx[value]
505  List<int> channelMessagesIdxs = new List<int>();
506 
507  // Create a list of messagesFromProcess
508  // The purpose of this list is to be the source of the messages that changed
509  // If we have a permutation we copy from this list to the MessageQ and
510  // don't have to worry about overriding messages in the messageQ
511  AttributeList messagesFromProcess = new AttributeList();
512  channelMessagesIdxs = ExstractChannelMessages(sourceProcessId, out messagesFromProcess);
513 
514  // Do the permutations
515  for (int idx = 0; idx < permutations.Count; idx ++)
516  {
517  if (permutations[idx] != idx)
518  {
519  int originalIndex = permutations[idx];
520  int finalIndex = idx;
521  this[channelMessagesIdxs[finalIndex]] = messagesFromProcess[originalIndex];
522  }
523  }
524

526  RetrieveFirstMessageFromQueue(ref ((BaseProcess)Element).MessageInProcess);
