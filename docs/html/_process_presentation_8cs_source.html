330  /*
331  * Events
332  * The events serve 3 processs :
333  * 1. ciircle movement
334  * 2. circle resize
335  * 3. selecting processes in add channel process
336  * 4. Selectng a process
337  * The following are the algorithms of the processes
338  * 1. Circle move
339  * Mouse doun - If the mouse button is the left button set a flag that we are in moving process
340  * Mouse move - If the flag is set move the circle to thenew location and update the textBlock and
341  * channels connected to the process
342  * Mouse Up - Update the network element with the new location and set the process as selectted
343  * 2. Circle resize
344  * Mouse down - if the button is the right button set a flag,get the distance of
345  * the mouse from the center of the circle
346  * Mouse move - each time :
347  * calculate the distance of the mouse from the center
348  * Add the difference of the distance from the distance calculated in the previouse step to the
349  * width and the height of the circle
350  * Redrow the circle, the textBlock and the channel connectted to the process
351  * Mouse Up - update the network elemet data
352  *3. Sellecting processes for adding a channel
353  * Mouse down - if in add chnnel process do nothing
354  * Mouse down - call main window add channel method with the process as a parameter
355  *4. Selecting a process - Is part of the moving circle process
356  */

428  public void Circle_MouseMove(object sender, System.Windows.Input.MouseEventArgs e)
429  {
430  var presentationElementEntry = presentationElements.First(pe => pe.Value.controls.Values.Any(control => control == sender));
431  if (MainWindow.ActivationPhase == MainWindow.ActivationPhases.Running)
432  {
433  return;
434  }
435 
436  Ellipse circle = (Ellipse)presentationElementEntry.Value.controls[PresentationElement.ControlKeys.Circle];
437  if (isInPlacingProcess== true && e.LeftButton == MouseButtonState.Pressed)
438  {
439  double left = e.GetPosition(canvas).X - circle.Width / 2;
440  double top = e.GetPosition(canvas).Y - circle.Height /2;
441  circle.Margin = new Thickness(left, top, 0, 0);
442  MoveTextBlockInCircle(presentationElementEntry.Value);
443  MoveChannels((BaseProcess)presentationElementEntry.Key);
444  SetBlinkingForAllProcesses();
445  }
446  if (inResizeProcess && e.RightButton == MouseButtonState.Pressed)
447  {
448  double distanceOfNewPointFromCircleCenter = DistanceFromCircleCenter(e.GetPosition(circle), circle);
449  double diffDistance = (distanceFromCenter - distanceOfNewPointFromCircleCenter) * 2;
450 
451  //If the movement of the cursor was too fast - stop the resize process
452  if (circle.Width + diffDistance < 0)
453  {
454  inResizeProcess = false;
455  UpdateNetworkElement(presentationElementEntry.Key);
456  mainWindow.UpdatePresentation();
457  mainWindow.SetSelectedProcess((BaseProcess)presentationElementEntry.Key);
458  return;
459  }
460  else
461  {
462  circle.Width = circle.Height = circle.Width + diffDistance;
463  }
464  //PrintCircleData();
465  SetTextBlockDimentions(presentationElementEntry.Value);
466  MoveTextBlockInCircle(presentationElementEntry.Value);
467  MoveChannels((BaseProcess)presentationElementEntry.Key);
468  distanceFromCenter = DistanceFromCircleCenter(e.GetPosition(circle), circle);
469  }
