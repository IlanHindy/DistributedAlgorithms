292  public static string RelativePath(string fromPath, string toPath)
293  {
294  string result = "";
295  int idx = 0;
296  string[] fromPathDirs = Regex.Split(fromPath, @":|\\");
297  string[] toPathDirs = Regex.Split(toPath, @":|\\");
298 
299  //If the paths are not from the same disk return the toPath
300  if (fromPath[0] != toPath[0])
301  {
302  return toPath;
303  }
304 
305  //Replace the common path start with ~
306  for (idx = 1; idx < Math.Min(fromPathDirs.Length, toPathDirs.Length); idx++)
307  {
308  if (fromPathDirs[idx] != toPathDirs[idx])
309  {
310  break;
311  }
312  }
313 
314  result += "~\\";
315  int commonPathLastIdx = idx;
316 
317  //ForEach directory left in the fromPath add "../"
318  for (idx = commonPathLastIdx - 1; idx < fromPathDirs.Length; idx++)
319  {
320  result += "..\\";
321  }
322 
323  //Add the rest of the toPath
324  for (idx = commonPathLastIdx; idx < toPathDirs.Length; idx++)
325  {
326  result += toPathDirs[idx];
327  result += "\\";
328  }
329 
330  return result;
331  }
