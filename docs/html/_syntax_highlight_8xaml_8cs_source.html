188  public static void HandleCollitions(List<TextFragment> list, TextFragment item)
189  {
190  int idx = 0;
191  while (idx < list.Count)
192  {
193 
194  int listItemStart = list[idx].index;
195  int listItemEnd = list[idx].index + list[idx].length - 1;
196  int itemStart = item.index;
197  int itemEnd = item.index + item.length - 1;
198 
199  // If the item is contained in the list item - discard the item and return
200  if (listItemStart <= itemStart && itemEnd <= listItemEnd)
201  {
202  return;
203  }
204 
205  // If the listItem is contained in the item - remove the listItem and continue
206  // (because there might be other listItem contained in item)
207  if (itemStart <= listItemStart && listItemEnd <= itemEnd)
208  {
209  list.RemoveAt(idx);
210  continue;
211  }
212 
213  // If they are foreign - continue to check the next item in the list
214  if (itemEnd < listItemStart || listItemEnd < itemStart)
215  {
216  idx++;
217  continue;
218  }
219 
220  // Else (partly collision) - discard the new item
221  return;
222  }
223 
224  // If we reached this point - the item is not contained or partly contained
225  // of any element in the list so we can add it to the list
226  list.Add(item);
227  }
228  }
