449  private bool SetKey()
450  {
451  dynamic key;
452  Type type = Type.GetType((string)ComboBox_SelectKeyType.SelectedItem);
453 
454  // If the type of the key is boolean or enum - get the key from the ComboBox
455  // and convert it from string to key
456  if (TextBox_InsertKey.Visibility == Visibility.Collapsed)
457  {
458  key = TypesUtility.GetKeyFromString(type,(string)ComboBox_SelectKey.SelectedItem);
459  }
460 
461  else if (type.Equals(typeof(string)))
462  {
463  key = TextBox_InsertKey.Text;
464  Microsoft.CSharp.CSharpCodeProvider csharpProvider = new Microsoft.CSharp.CSharpCodeProvider();
465  if (!csharpProvider.IsValidIdentifier(key))
466  {
467  CustomizedMessageBox.Show("Error in key insertion :\n" +
468  "The value : '" + TextBox_InsertKey.Text + " is not a leagal c# variable name",
469  "Create Attribute Window", Icons.Error);
470  return false;
471  }
472  }
473 
474  // If the type of the key is other Try to convert the string in the TextBox to
475  // the selected type - if failed - return false
476  else
477  {
478  bool converted;
479  key = TypesUtility.ConvertValueFromString(type, TextBox_InsertKey.Text, out converted);
480  if (!converted)
481  {
482  CustomizedMessageBox.Show("Error in key insertion :\n" +
483  "The value : '" + TextBox_InsertKey.Text + "' Cannot be converted to type : " + type.ToString(),
484  "Create Attribute Window", Icons.Error);
485  return false;
486  }
487  }
488 
489  // Check if the key already exist the dictionary (If the attribute should be inserted
490  // to a list no check is made)
491  // If the check succeeded set the member and return true
492  if (CheckIfKeyExist(key))
493  {
494  this.key = key;
495  return true;
496  }

498  {
499  return false;
