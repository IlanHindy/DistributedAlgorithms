474  private Attribute EvaluateParameter(EvaluationMode evaluationMode, BaseNetwork network, BaseProcess process, BaseChannel channel, BaseMessage message, Attribute parameter)
475  {
476  if (parameter.Value.GetType().IsEnum)
477  {
478  if (!NetworkElementExist(network, process, channel, message))
479  {
480  return new Attribute() { Value = false };
481  }
482  Attribute attribute = FindAttribute(network, process, channel, message, parameter.Value);
483  if (attribute == null)
484  {
485  string typeOfHostingElementStr = TypesUtility.GetKeyToString(or[brp.ork.HostingElementType]);
486  throw new EvaluationException("Could not find attribute " + TypesUtility.GetKeyToString(parameter.Value) + " In element of type " + typeOfHostingElementStr);
487  }
488  else
489  {
490  if (Attribute.GetValueCategory(attribute) == Attribute.AttributeCategory.ListOfAttributes)
491  {
492  if (or[brp.ork.ListParameterOneOrAll] == ListParameterOneOrAll.One)

495  int requestedIndex = or[brp.ork.IndexInList];
496  if (elements.Count() <= requestedIndex)

499  }
500  else
501  {
502  if (Attribute.GetValueCategory(attribute) == Attribute.AttributeCategory.ListOfAttributes)
503  {
504  return (Attribute)elements[requestedIndex];
505  }
506  else
507  {
508  return new Attribute() { Value = elements[requestedIndex] };
509  }
510  }
511  }
512  }
513  return attribute;
514  }
515  }
516  if (parameter.Value.GetType().Equals(typeof(Breakpoint)))
517  {
518  return new Attribute() { Value = parameter.Value.Evaluate(evaluationMode, network, process, channel, message) };
519  }

535  private bool Evaluate(AttributeList parameters)
536  {
537  Operators Operator = or[brp.ork.Operator];
538  switch (Operator)
539  {
540  case Operators.None: return false;
541  case Operators.OperationsOnOtherElements: return Evaluate(parameters, (x, y) => { return EvaluateForOtherElements(x, y); });
542  case Operators.DataEqual: return Evaluate(parameters, (x, y) => { return (x is NetworkElement)? x.EqualsTo(y):x.Equals(y); });
543  case Operators.DataNotEqual: return Evaluate(parameters, (x, y) => { return (x is NetworkElement) ? !(x.EqualsTo(y)) : !(x.Equals(y)); });
544  case Operators.And: return Evaluate(parameters, (x, y) => { return x && y; });
545  case Operators.Bigger: return Evaluate(parameters, (x, y) => { return x > y; });
546  case Operators.BiggerOrEqual: return Evaluate(parameters, (x, y) => { return x >= y; });
547  case Operators.Equeal: return Evaluate(parameters, (x, y) => { return x == y; });
548  case Operators.NotEqual: return Evaluate(parameters, (x, y) => { return x != y; });
549  case Operators.Not: return Evaluate(parameters, (x, y) => { return !x; });
550  case Operators.Or: return Evaluate(parameters, (x, y) => { return x || y; });
551  case Operators.SmalerOrEqual: return Evaluate(parameters, (x,y) => {return x<=y;});
552  case Operators.Smaller: return Evaluate(parameters, (x, y) => { return x < y; });
553  default: return false;
554  }
