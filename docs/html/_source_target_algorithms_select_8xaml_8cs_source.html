93  public SourceTargetAlgorithmsSelect()

95  InitializeComponent();
96  List<SelectControlData> data = new List<SelectControlData>();
97 
98  // Create a list of the subjects and an enable list for them
99  List<string> subjects = TypesUtility.GetAlgorithmsSubjects();
100  var enableSubjectsSource = Enumerable.Repeat(true, subjects.Count).ToList();
101 
102 
103  // Create the data for the source subject control
104  // For the source subject the System subject is disabled (It cannot be used as source)
105  enableSubjectsSource[subjects.IndexOf("Base")] = false;
106  List<int> initialySelectedSubject = new List<int> { subjects.IndexOf("Templates") };
107  data.Add(new SelectControlData() { options = subjects, enableItems = enableSubjectsSource, initiallySelected = initialySelectedSubject });
108 
109  // Create the data for the control of the source algorithm
110  data.Add(new SelectControlData { options = new List<string>() });
111 
112  // Create the data for the control of the target subject
113  // For the target subject the System and Templates subjects are diasabled
114  List<string> allSubjects = subjects.Concat(Config.Instance.GetSubjects().ToList()).ToList();
115  allSubjects = allSubjects.Distinct().ToList();
116  var enableSubjectsTragets = Enumerable.Repeat(true, allSubjects.Count).ToList();
117  enableSubjectsTragets[allSubjects.IndexOf("Base")] = false;
118  enableSubjectsTragets[allSubjects.IndexOf("Templates")] = false;
119  data.Add(new SelectControlData { options = allSubjects, enableItems = enableSubjectsTragets });
120 
121  // Create the data for the control of the target algorithm
122  data.Add(new SelectControlData { options = new List<string>() });
123  this.data = data;
124 
125  // Order of changing the presentation according to selection change is as follows
126  // Source subject change :
127  // The Source algorithms control change (To present the algorithms of the subject)
128  // The Target subject control change (To set the Target subject is set to the source subject
129  // Target subject change :
130  // The Target algorithms control change (To present the algorithms of the subject)
131  //
132  // There for the order of initialize of the controls is in reverse order
133  // (So the initiation will not override the data that was set by a control that is before
134  // in the change chain)
135  Selection_TargetAlgorithm.Init(data[3]);
136  Selection_TargetSubject.Init(data[2]);
137  Selection_SourceAlgorithm.Init(data[1]);
138  Selection_SourceSubject.Init(data[0]);
139  }
