347  private void Button_Select_Click(object sender, RoutedEventArgs e)
348  {
349  string subject = Selection_Subject.GetSelection().selectionText;
350  string algorithm = Selection_Algorithm.GetSelection().selectionText;
351  string network = Selection_Network.GetSelection().selectionText;
352  string file = Selection_File.GetSelection().selectionText + fileExtension;
353 
354  // If the network is new the dictionary of the network is not found in the config
355  // so we generate default path
356  string path;
357  try
358  {
359  path = Config.Instance[Config.Keys.Subjects][subject][algorithm][Config.AlgorithmKeys.Networks][network][pathKey];
360  }
361  catch
362  {
363  path = Config.Instance.GenerateDataFilePath(subject, algorithm, network);
364  }
365 
366  string absPath = System.IO.Path.GetFullPath(path);
367 
368  switch (selectSource)
369  {
370  case SelectSource.New:
371  result = EndSelectionForNew(absPath, file);
372  break;
373  case SelectSource.Open:
374  result = EndSelectionForOpen(absPath, file);
375  break;
376  case SelectSource.SaveAs:
377  result = EndSelectionForSaveAs(absPath, file);
378  break;
379  case SelectSource.SaveDebug:
380  result = EndSelectionForSaveAs(absPath, file);
381  break;
382  case SelectSource.Debug:
383  result = EndSelectionForDebug(absPath, file);
384  break;
385  case SelectSource.Log:
386  result = SelectResult.Log;
387  break;
388  default:
389  result = SelectResult.Quit;
390  break;
391  }
392 
393  switch (result)
394  {
395  case SelectResult.Quit:
396  Close();
397  break;
398  case SelectResult.Cancel:
399  break;
400  case SelectResult.Debug:
401  Config.Instance.UpdateSelectedDebugFileChanged(subject, algorithm, network, file, path);
402  Close();
403  break;
404  case SelectResult.Log:
405  Config.Instance[Config.Keys.SelectedLogFileName] = file;
406  Close();
407  break;
408  default:
409  if (selectSource == SelectSource.SaveDebug)
410  {
411  Config.Instance.UpdateSelectedDebugFileChanged(subject, algorithm, network, file, path);
412  }
413  else
414  {
415  Config.Instance.UpdateSelectedDataFileChanged(subject, algorithm, network, file, path);
416  }
417  Close();
418  break;
419  }
420  }
