430  private void ReOrder(Label targetLabel)
431  {
432  // Get the rows of the source and target labels
433  int sourceLabelRow = Grid.GetRow(sourceLabel);
434  int targetLabelRow = Grid.GetRow(targetLabel);
435 
436  // If the source and target rows are the same - exit
437  if (sourceLabelRow == targetLabelRow)
438  {
439  return;
440  }
441 
442  int sourcePermutation = permutations[sourceLabelRow];
443 
444  // If the sourceLabelRow is smaller than the targetLabelRow that means that the
445  // drag is moving downwards The following is an example of what is done:
446  // Original list:
447  // [0] text0
448  // [1] text1
449  // [2] text2
450  // [3] text3
451  // Drag results:
452  // source row : 0
453  // target row : 2
454  // Transformation step 1
455  // [0] is set to 1 (row = 1)
456  // [1] is set to 2 (row = 2)
457  // The result of this step is [1,2,2,3]
458  // Transformation step 2
459  // Move the source row to the target index
460  // The result of this step is [1,2,0,3]
461 
462  if (sourceLabelRow < targetLabelRow)
463  {
464  for (int row = sourceLabelRow + 1; row <= targetLabelRow; row ++)
465  {
466  permutations[row - 1] = permutations[row];
467  }
468  }
469 
470  // If the sourceLabelRow is smaller than the targetLabelRow that means that the
471  // drag is moving downwards The following is an example of what is done:
472  // Original list:
473  // [0] text0
474  // [1] text1
475  // [2] text2
476  // [3] text3
477  // Drag results:
478  // source row : 2
479  // target row : 0
480  // Transformation step 1
481  // [2] is set to 1 (row = 1)
482  // [1] is set to 0 (row = 0)
483  // The result of this step is [0,0,1,3]
484  // Transformation step 2
485  // Move the source row to the target index
486  // The result of this step is [2,0,1,3]
487  else if (sourceLabelRow > targetLabelRow)
488  {
489  for(int row = sourceLabelRow - 1; row >= targetLabelRow; row --)
490  {
491  permutations[row + 1] = permutations[row];
492  }
493  }
494  permutations[targetLabelRow] = sourcePermutation;
495 
496  // Move the texts according to the permutation
497  SetContents();
498  sourceLabel = targetLabel;
499  }
